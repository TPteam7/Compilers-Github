int add(int a, int b) {
    int sum;
    int doubleSum;
    int temp;

    sum = a + b;  // Compute sum
    temp = sum + 5;  // Add constant
    doubleSum = temp * 2;  // Double the sum

    if(doubleSum > 50) {
        print(doubleSum);
    }
    else {
        print(0);
    }

    return doubleSum;
}

int subtract(int a, int b) {
    int c = a - b;

    while(c < 20) {
        print(c);
        c = c + 2;
    }

    return c;
}

int factorial(int n) {
    int result = 1; 
    int current = n; 
   
    while (current > 1) {
        result = result * current; 
        current = current - 1;      
    }
 
    return result;
}
